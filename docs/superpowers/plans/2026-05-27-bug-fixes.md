```python
def _check_gene2go_has_taxid(self, taxid: int, go_version: str = None) -> bool:
    """检查 gene2go.gz 是否包含指定 taxid的数据

    全文件扫描以确保找到所有物种（包括人类等大数据集）。

    Args:
        taxid: NCBI Taxonomy ID

    Returns:
        True 如果找到匹配的 taxid
    """
    if go_version is None:
        downloader = DataDownloader(root_dir=str(self.root_dir))
        go_version = downloader.get_latest_go_version()

    gene2go_path = self.root_dir / "basic" / "go" / go_version / "gene2go.gz"

    if not gene2go_path.exists():
        return False

    with gzip.open(gene2go_path, "rt", encoding="utf-8") as fh:
        for line in fh:
            line = line.strip()
            if not line or line.startswith("#"):
                continue
            parts = line.split("\t")
            if len(parts) < 1:
                continue
            try:
                if int(parts[0]) == taxid:
                    return True
            except (ValueError, IndexError):
                continue

    return False
```

```python
if gene_info_path.exists():
```

- [ ] **Step 2: 添加自动查找物种子目录的逻辑**

在 `DatabaseManager.__init__()` 或 `load_database()` 中添加：

```python
def _find_species_dir(self, database_dir: Path, species: str) -> Path:
    """自动查找物种数据库目录

    搜索 database_dir 下的 organism/v*/{species}/ 子目录

    Args:
        database_dir: 数据库根目录
        species: 物种代码

    Returns:
        物种数据库目录路径
    """
    # 模式 1: database/organism/v{date}/{species}/
    organism_dir = database_dir / "organism"
    if organism_dir.exists():
        for version_dir in sorted(organism_dir.iterdir(), reverse=True):
            if version_dir.is_dir():
                species_dir = version_dir / species
                if species_dir.exists():
                    return species_dir

    # 模式 2: 直接在 database_dir 下
    if (database_dir / f"{species}.GO2gene.tab.gz").exists():
        return database_dir

    # 回退: 返回原目录
    return database_dir
```
